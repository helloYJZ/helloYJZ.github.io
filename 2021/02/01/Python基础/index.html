# try...except...
import math
for i in range(10):
    try:
        input_number = input("enter a number: ") # input()让用户输入
        if input_number == 'q':
            break
        result = math.log(float(input_number))
        print(result)
    except ValueError: # except可以写多个
        print('ValueError: input must > 0')
    except ZeroDivisionError:
        print('divided by 0')
    except Exception: 
        print('Error')
    finally:
        print('always print finally block') # finally部分始终会执行

继承

# 定义父类
class Parent:
    number = 100
    def __init__(self):
        print('调用父类构造函数')
    def parentMethod(self):
        print('调用父类方法')
    def setAttr(self,attr):
        Parent.parentAttr = attr
    def getAttr(self):
        print('父类属性：', Parent.parentAttr)
    def newMethod(self):
        print('父类要被重写的方法')
# 定义子类
class child(Parent):
    def __init__(self):
        print('调用子类构造函数')
    def childMethod(self):
        print('调用子类方法')
    def newMethod(self): # 子类方法重新，名字和父类一致
        print('子类把父类的方法改掉了')

# 实例化子类
c = child() # '调用子类构造函数'
c.childMethod() # '调用子类方法'
c.parentMethod() # '调用父类方法'
c.setAttr(100)
c.getAttr() # '父类属性：100'
c.newMethod() # '子类把父类的方法改掉了'
